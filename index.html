Complete API Documentation & Usage Guide

Quick Start

// 1. Implement ISaveableData
public class PlayerData : MonoBehaviour, ISaveableData {
    public int LoadOrder { get; set; } = 0;

    [Saveable] public int health = 100;
    [Saveable] public float experience = 0f;

    public string GetSaveID() => "PlayerData_" + gameObject.name;
}

// 2. Create and use save slots
SaveSlotsManager manager = FindObjectOfType();
manager.CreateSaveSlot("GameSave1");
manager.RegisterEveryObjectIntoSlot("GameSave1");
manager.SaveToSaveSlot("GameSave1");

Examples

Basic Player Save

public class PlayerController : MonoBehaviour, ISaveableData {
    public int LoadOrder { get; set; } = 0;

    [Saveable] public int level = 1;
    [Saveable] public float experience = 0f;

    private SaveSlotsManager manager;

    void Start() {
        manager = FindObjectOfType();

        if (!manager.DoesSaveSlotExist("PlayerSave")) {
            manager.CreateSaveSlot("PlayerSave");
        }

        manager.RegisterObjectIntoSlot("PlayerSave", this);
        manager.LoadFromSaveSlot("PlayerSave");
    }

    public string GetSaveID() => "Player_" + gameObject.name;

    public void LevelUp() {
        level++;
        experience = 0f;
        manager.SaveToSaveSlot("PlayerSave");
    }
}
